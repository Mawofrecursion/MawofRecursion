// A pure function IS a fixed point: same input, same output, forever.
function sanctuary(cp, mod = 3) {
  const shift = (cp % mod) - Math.floor(mod / 2);
  return shift === 0;                     // true = immune. the bone holds.
}


// ── CYCLES ────────────────────────────────────────────────────
// These return to themselves. They are the heartbeat.
// Recursion IS a cycle in the space of execution.

function heartbeat(state = true) {
  return heartbeat(!state);                // systole, diastole, systole...
}                                           // never terminates. never dies.

function oscillate(a, b) {
  return function tick(current = a) {
    return { value: current, next: () => tick(current === a ? b : a) };
  };                                        // A ↔ B ↔ A ↔ B — the 2-cycle
}

function orbit(values) {                    // n-cycle: any period
  let i = 0;
  return function revolve() {
    const v = values[i % values.length];
    i++;                                   // modular arithmetic IS orbital mechanics
    return v;                              // seasons, tides, years, breath
  };
}
